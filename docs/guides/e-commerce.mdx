```javascript Node.js
class CatalogScraper {
  constructor(apiKey) {
    this.client = new WebLinq(apiKey);
  }

async scrapeCatalog(url, options = {}) {
const { maxPages = 1 } = options;
const products = [];

    for (let page = 1; page <= maxPages; page++) {
      const pageUrl = this.getPageUrl(url, page);
      const pageProducts = await this.scrapePage(pageUrl);
      products.push(...pageProducts);

      // Check if next page exists
      if (!await this.hasNextPage(pageUrl)) break;
    }

    return products;

}

async scrapePage(url) {
const response = await this.client.extract({
url,
selectors: {
products: {
selector: '.product-item',
type: 'list',
data: {
name: '.product-name',
price: '.product-price',
image: {
selector: '.product-image',
attr: 'src'
},
url: {
selector: '.product-link',
attr: 'href'
}
}
}
}
});

    return response.products;

}

async hasNextPage(url) {
const response = await this.client.extract({
url,
selectors: {
nextPage: {
selector: '.pagination .next',
attr: 'href'
}
}
});

    return !!response.nextPage;

}

getPageUrl(baseUrl, page) {
const url = new URL(baseUrl);
url.searchParams.set('page', page);
return url.toString();
}
}

// Usage
const scraper = new CatalogScraper('YOUR_API_KEY');

const products = await scraper.scrapeCatalog('https://example.com/products', {
maxPages: 5
});

console.log(`Scraped ${products.length} products`);

````

```python Python
class CatalogScraper:
    def __init__(self, api_key: str):
        self.client = WebLinq(api_key)

    async def scrape_catalog(self, url: str, **options) -> List[dict]:
        max_pages = options.get('max_pages', 1)
        products = []

        for page in range(1, max_pages + 1):
            page_url = self.get_page_url(url, page)
            page_products = await self.scrape_page(page_url)
            products.extend(page_products)

            # Check if next page exists
            if not await self.has_next_page(page_url):
                break

        return products

    async def scrape_page(self, url: str) -> List[dict]:
        response = await self.client.extract({
            'url': url,
            'selectors': {
                'products': {
                    'selector': '.product-item',
                    'type': 'list',
                    'data': {
                        'name': '.product-name',
                        'price': '.product-price',
                        'image': {
                            'selector': '.product-image',
                            'attr': 'src'
                        },
                        'url': {
                            'selector': '.product-link',
                            'attr': 'href'
                        }
                    }
                }
            }
        })

        return response['products']

    async def has_next_page(self, url: str) -> bool:
        response = await self.client.extract({
            'url': url,
            'selectors': {
                'next_page': {
                    'selector': '.pagination .next',
                    'attr': 'href'
                }
            }
        })

        return bool(response['next_page'])

    def get_page_url(self, base_url: str, page: int) -> str:
        from urllib.parse import urlparse, parse_qs, urlencode

        parsed = urlparse(base_url)
        query = parse_qs(parsed.query)
        query['page'] = [str(page)]

        return parsed._replace(query=urlencode(query, doseq=True)).geturl()

# Usage
async def main():
    scraper = CatalogScraper('YOUR_API_KEY')

    products = await scraper.scrape_catalog(
        'https://example.com/products',
        max_pages=5
    )

    print(f"Scraped {len(products)} products")

asyncio.run(main())
````

### Rate Limiting

<Warning>Always respect rate limits and implement proper delays between requests.</Warning>

```javascript
class RateLimiter {
  constructor(requestsPerSecond) {
    this.requestsPerSecond = requestsPerSecond;
    this.queue = [];
    this.processing = false;
  }

  async add(fn) {
    return new Promise((resolve, reject) => {
      this.queue.push({ fn, resolve, reject });
      if (!this.processing) this.process();
    });
  }

  async process() {
    this.processing = true;

    while (this.queue.length > 0) {
      const { fn, resolve, reject } = this.queue.shift();

      try {
        const result = await fn();
        resolve(result);
      } catch (error) {
        reject(error);
      }

      await new Promise((resolve) => setTimeout(resolve, 1000 / this.requestsPerSecond));
    }

    this.processing = false;
  }
}
```

### Error Handling

```javascript
class ErrorHandler {
  constructor() {
    this.retryCount = 3;
    this.retryDelay = 1000;
  }

  async withRetry(fn) {
    for (let i = 0; i < this.retryCount; i++) {
      try {
        return await fn();
      } catch (error) {
        if (i === this.retryCount - 1) throw error;

        await new Promise((resolve) => setTimeout(resolve, this.retryDelay * Math.pow(2, i)));
      }
    }
  }
}
```
